```
tar -xzf mydumper-<version>-linux.tar.gz
cd 解压后的文件夹
mkdir build
cd build
cmake ..
make
sudo make install
```



```
[root@localhost build]# mydumper --version
mydumper v0.16.9-1, built against MariaDB 10.3.6 with SSL support
```
